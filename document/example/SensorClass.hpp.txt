#pragma once
#include "driver/i2c_master.h"


class I2C_BUS{
    private:
        i2c_master_bus_config_t bus_cfg ={};
        i2c_master_bus_handle_t bus_handle = nullptr;
    I2C_BUS(i2c_port_num_t p_port,gpio_num_t p_sda,gpio_num_t p_scl){
        bus_cfg.i2c_port      = p_port;
        bus_cfg.sda_io_num    = p_sda;
        bus_cfg.scl_io_num    = p_sda;
        bus_cfg.clk_source  = I2C_CLK_SRC_DEFAULT;
        bus_cfg.glitch_ignore_cnt = 1;
        bus_cfg.intr_priority = 0;
        bus_cfg.flags.enable_internal_pullup = true;
        bus_cfg.flags.allow_pd = false; 
    }    
    void initialize(){
        i2c_new_master_bus(&bus_cfg,&bus_handle);
    }
    i2c_master_bus_handle_t getHandle(){
        return bus_handle;
    }
};
